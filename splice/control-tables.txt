	TABLE_ID BIGINT NOT NULL,
	SCHEMANAME VARCHAR(200),
	TABLENAME VARCHAR(200),
	TABLE_OWNER VARCHAR(20),
	SPLITCOLUMN VARCHAR(200),
	MAPPER VARCHAR(10),
	LOADTYPE CHAR(1),
	SOURCE_TYPE CHAR(1),

	DIRECTION CHAR(1),
	TABLE_ACTIVE_FLAG CHAR(1),
	SQOOPREQUIRED CHAR(1),

	ORDER_BY SMALLINT,

	CREATED_BY VARCHAR(200) DEFAULT 'SPLICE' NOT NULL,
	MODIFIED_DATE TIMESTAMP DEFAULT current_timestamp NOT NULL,
	MODIFIED_BY VARCHAR(200) DEFAULT 'SPLICE' NOT NULL,
	HIVE_DIFF CHAR(1) DEFAULT 'N',
